‚úÖ Spy using cy.intercept

it('should display a list of todos', () => {
  cy.intercept('GET', '/todos').as('todos')
  cy.visit('/')
  cy.wait('@todos')

})

‚úÖ Spread to handle multiple requests

cy.intercept('GET', '/users').as('users')

  (todos, users, comments) => {
    // each interception is now an individual argument
  }
)
